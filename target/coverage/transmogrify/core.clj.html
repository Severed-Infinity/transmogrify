001  (ns transmogrify.core
002    (:require [transmogrify.internal :refer :all]
003              [transmogrify.wip :refer :all]
004              [clojure.spec.alpha :as spec]
005              [clojure.spec.test.alpha :as stest]
006              [spec-tools.spec :as st]))
007  
008  (defn transmogrify 
009    "(transmogrify ::html [:html [:body [:p \"hello\"]]])
010      => <!DOCTYPE html>
011         <html>
012           <body>
013             <p>hello</p>
014           </body>
015         </html>"
016    [language & data]
017    (comment 
018      "from the data we want an element of which is a tag and content a.k.a data/element"
019      "we recurively transform the data and generate the apprioate file with contents"
020      "try aim for multiple file transformation/optimisation outputs?"
021      "take a leading tag as file name in datastructure?"
022      (let [transformed-data (let [element (first data)
023                                   tag (first element)
024                                   content (second element)]
025                               (if (!= content element?)
026                                 (transfrom-to-language tag content)
027                                 (recur language content)))]
028        (output-to-file language transformed-data))) 
029    
030    (comment "using language argument lookup the respective algorithm"
031      "the actual work is done by the transmogrifier")
032    (transmogrifier language data))
033  
034  (spec/fdef
035    transmogrify
036    :args (spec/cat :language keyword? :data coll?)
037    :ret str)
038  
039  (transmogrify ::html [:html])
040  
041  
042  #_(defn -main []
043                (println "Hello, World!"))
