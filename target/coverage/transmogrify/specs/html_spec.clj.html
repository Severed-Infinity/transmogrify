192      (defn- html-spec
193        ([html-tag
194          (let [keyword-html-tag (keyword (name html-tag))]
195            (s/or
196              :map-form (st-ds/spec html-tag {:tag (tag keyword-html-tag)})
197              :vector-form (s/cat :tag (tag keyword-html-tag))))])
198  
199        ([html-tag content
200          (let [keyword-html-tag (keyword (name html-tag))
201                required-content (:req content)
202                optional-content (:opt content)]
203            (s/or
204              :map (st-ds/spec html-tag {:tag     (tag keyword-html-tag)
205                                         :content (s/cat)})))]))
