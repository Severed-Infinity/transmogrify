030  (defn- html-styles->str [styles]
031    (str/join (map (fn [[k v]] (str (name k) ":" v ";")) styles)))

033  (defn- html-attrs->str [attrs]
034    (str/join
035      (map
036        (fn [[k v]]
037          (str
038            (str/lower-case (name k)) 
039            "=\""
040            (cond
041              (map? v) (html-styles->str v)
042              (sequential? v) (str/join " " v)
043              :else v)
044            "\" "))
045        attrs)))

053    [element]

062  (defn- void-tag?
063    [tag]
064    (let [tag-type (spec/or :void ::html/void-elements)           
065          conformed-tag (spec/conform tag-type tag)]
066      (if (spec/valid? tag-type tag)
067        (case (first conformed-tag)
068          :void true
069          :unescapable false)
070        false)))
